  • private static int partition(int[] items,int low, int high)
{

    int i=0;
    int j=items.length-1;
    int k=items.length-1;
    int pivot=items[low];
    while(j < items.length)
    {
        if (items[j] > pivot)
        {
            swap(j,i,items);
            j++;
            i++;				
        }
        else if(items[j] < pivot)
        {
            j++;
        }
        else
        {
            swap(j,k-1,items);
            k--;
        }
    }
    return i;
}


private static void swap(int i,int j,int[] items)
{
    int temp=items[i];
    items[i]=items[j];
    items[j]=temp;
}
